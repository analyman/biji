% config some language in listings package


%{{{ Macro for building new lstenv, short and long style
%{{{ for short style listings
\long\def\new@lstenv@short#1#2{\lstnewenvironment{#1}[1][]
% before
{% lstset
\lstset{language={#2},%
label=##1,%
basicstyle=\ttfamily\small,%
showtabs=false, showspaces=false, showstringspaces=false, tabsize=4,%
frame=trbl, frameround=tttt, %
backgroundcolor=\color{white}}
}
% after
{}
}
%}}}

%{{{ for long style listings
\definecolor{color@codebg}{rgb}{0.95, 0.98, 0.98}
\definecolor{color@number}{rgb}{0.4, 0.2, 0.3}
\long\def\new@lstenv@long#1#2{\lstnewenvironment{#1}[1][]
% before
{% lstset
\lstset{language={#2},%
        label=#1,%
        numbers=left, numberstyle=\tt\color{color@number}, stepnumber=1, numbersep=8pt,%
        showtabs=false, showspaces=false, showstringspaces=false, tabsize=4,%
        frame=trBL, frameround=fttt,%
        backgroundcolor=\color{color@codebg}}
}
% after
{}
}
%}}}
%}}}

% clang
\subimport{.tex/listings/}{clang.tex}
\new@lstenv@short{sccode}{MYCPP}
\new@lstenv@long{clangcode}{MYCPP}

% tex and others
\subimport{.tex/listings/}{tex.tex}
